      while (el && el.tagName !== "H3") el = el.previousElementSibling;
      if (el) el.style.display = allHidden ? "none" : "";
    });
  }

  let active = null;

  const pills = Array.from(allTags).sort().map(tag => {
    const btn = document.createElement("button");
    btn.textContent = tag;
